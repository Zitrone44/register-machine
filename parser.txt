start = Automate
Automate = commandLine:CommandLine+
CommandLine = command:Command " " parameter:Parameter '\n' {return {command, parameter}}
Command = command:("Load"/"Store"/"Add"/"Sub"/"Mul"/"Div"/"Goto"/"JZero")
Parameter = NumParameter/VarParameter
NumParameter = num:([1-9][0-9]*) {return num.join('')}
VarParameter = "[" varName:NumParameter "]" {return "[" + varName + "]"}
